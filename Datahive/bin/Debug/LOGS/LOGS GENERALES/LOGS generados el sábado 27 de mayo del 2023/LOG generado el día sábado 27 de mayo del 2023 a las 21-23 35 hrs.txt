SELECT r.RDB$FIELD_NAME AS NOMBRE_CAMPO, 
 CASE f.RDB$FIELD_TYPE  WHEN 261 THEN 'TEXT'  WHEN 14 THEN 'CHAR' WHEN 40 THEN 'VARCHAR'  WHEN 27 THEN 'DOUBLE' WHEN 10 THEN 'FLOAT'  WHEN 16 THEN 'BIGINT'  WHEN 8 THEN 'INT'   WHEN 9 THEN 'BIGINT' WHEN 7 THEN 'SMALLINT' WHEN 12 THEN 'DATE'  WHEN 13 THEN 'TIME'    WHEN 35 THEN 'TIMESTAMP'    WHEN 37 THEN 'VARCHAR'   ELSE 'UNKNOWN'  END AS TIPO_DE_DATO,
CASE f.RDB$FIELD_TYPE  WHEN 14 THEN '(' || f.RDB$FIELD_LENGTH || ')'   WHEN 37 THEN '(' || f.RDB$FIELD_LENGTH || ')'   WHEN 8 THEN ''   WHEN 9 THEN '(' || f.RDB$FIELD_LENGTH || ')'   WHEN 35 THEN '(' || f.RDB$FIELD_LENGTH || ')'   WHEN 10 THEN '(' || f.RDB$FIELD_PRECISION || ',' || f.RDB$FIELD_SCALE || ')'   WHEN 11 THEN '(' || f.RDB$FIELD_PRECISION || ',' || f.RDB$FIELD_SCALE || ')'   ELSE ''  END AS LONGITUD,
CASE WHEN r.RDB$NULL_FLAG = 1 THEN 'NOT NULL' ELSE '' END AS NOT_NULL,
CASE WHEN s.RDB$FIELD_NAME IS NOT NULL THEN 'PRIMARY KEY' ELSE '' END AS PRIMARY_KEY
FROM RDB$RELATION_FIELDS r
LEFT JOIN RDB$FIELDS f ON r.RDB$FIELD_SOURCE = f.RDB$FIELD_NAME
LEFT JOIN RDB$COLLATIONS coll ON f.RDB$COLLATION_ID = coll.RDB$COLLATION_ID
LEFT JOIN RDB$CHARACTER_SETS cset ON f.RDB$CHARACTER_SET_ID = cset.RDB$CHARACTER_SET_ID
LEFT JOIN RDB$RELATION_CONSTRAINTS rc ON r.RDB$RELATION_NAME = rc.RDB$RELATION_NAME AND rc.RDB$CONSTRAINT_TYPE = 'PRIMARY KEY'
LEFT JOIN RDB$INDEX_SEGMENTS s ON rc.RDB$INDEX_NAME = s.RDB$INDEX_NAME AND r.RDB$FIELD_NAME = s.RDB$FIELD_NAME
WHERE r.RDB$RELATION_NAME='Table01'ORDER BY r.RDB$FIELD_POSITION;
